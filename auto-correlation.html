import numpy as np

# Parameters
AVG_RET = 0
RET_STD = 0.01
NUMBER_OF_BARS = 1_000_000
MIN_HOLDING_PERIOD = 1
MAX_HOLDING_PERIOD = 250
BASELINE_BARS = 1

np.random.seed(1234)
instrument_ret = pd.Series(np.random.normal(AVG_RET, RET_STD, NUMBER_OF_BARS))
adjustment = (1 / np.prod(1 + instrument_ret))**(1 / NUMBER_OF_BARS) - 1
instrument_ret += adjustment  # remove bias
instrument_prc = (1 + instrument_ret).cumprod()
baseline_std = instrument_prc.pct_change(BASELINE_BARS).rolling(30).std()

data = []
for holding_period in range(MIN_HOLDING_PERIOD, MAX_HOLDING_PERIOD + 1):
    expected_trade_ret_std = baseline_std * np.sqrt(holding_period / BASELINE_BARS)
    trade_ret = instrument_prc.pct_change(holding_period).shift(-holding_period).dropna()
    trade_sigma = trade_ret / expected_trade_ret_std
    pct_outliers = sum(trade_sigma.abs() > 2) / len(trade_ret)
    data.append([holding_period, pct_outliers])
data = pd.DataFrame(data, columns=("holding_period", "pct_outliers")).set_index("holding_period")

def remove_market_bias(prices: pd.Series) -> pd.Series:
    # Only works for continuous futures built from backadjusting/difference method!
    bias = prices.iat[-1] - prices.iat[0]
    bars = len(prices)
    avg_bias = bias / bars
    price_changes = prices.diff().fillna(0) - avg_bias
    return price_changes.cumsum()
